            if (select) select.innerHTML = '';

        function updateFinalPrice() {
            const prodSelect = document.getElementById('order-product');
            if (!prodSelect || !prodSelect.value) return;
            
            const prodId = prodSelect.value;
            const qtyField = document.getElementById('order-qty');
            const qty = parseInt(qtyField ? qtyField.value : 1) || 1;
            const producto = inventario.find(p => p.id == prodId);
            
            if(producto) {
                const precioPublico = producto.precioTienda * 1.20;
                const totalField = document.getElementById('order-total');
                if (totalField) totalField.value = `$${(precioPublico * qty).toFixed(2)}`;
            }
        }

        function processOrder(isAccepted) {
            if (!isAccepted) {
                alert("Puedes editar la información en el formulario.");
                return;
            }

            const name = document.getElementById('order-name').value.trim();
            const address = document.getElementById('order-address').value.trim();
            const email = document.getElementById('order-email').value.trim();
            const prodId = document.getElementById('order-product').value;
            const qty = parseInt(document.getElementById('order-qty').value);
            const datetime = document.getElementById('order-datetime').value;

            if(!name || !address || !email) {
                alert("Por favor completa todos los campos del cliente.");
                return;
            }

            const producto = inventario.find(p => p.id == prodId);
            const totalStr = document.getElementById('order-total').value;

            if (producto.stock < qty) {
                alert(`Lo sentimos, solo quedan ${producto.stock} unidades de este producto.`);
                return;
            }
            
            producto.stock -= qty;
            
            ordenesCompra.push({
                id: Date.now(), 
                cliente: name,
                email: email, 
                producto: producto.nombre,
                cantidad: qty,
                total: totalStr,
                direccion: address,
                estado: "Pendiente"
            });

            renderAdminData(true); 

            const templateParams = {
                client_name: name,
                client_email: email,
                product_name: producto.nombre,
                quantity: qty,
                total_price: totalStr,
                delivery_address: address,
                delivery_date: datetime
            };

            emailjs.send(EMAILJS_SERVICE_id, EMAILJS_TEMPLATE_CLIENTE, templateParams)

                    alert(`¡Orden Procesada Exitosamente!\nSe ha enviado el comprobante real al correo: ${email}.`);

                    alert("La orden se guardó localmente, pero falló el envío del correo de confirmación.");
                });

            document.getElementById('order-form').reset();
            initClientView();
        }

                alert(`Producto ${prod.nombre} actualizado correctamente.`);
                saveData();
                renderAdminData(true); 
            }
        }

        function enviarCorreoAdminStockBajo(nombreProducto, stockActual) {
            const adminEmail = "playnow2026@gmail.com"; 
            
            const adminParams = {
                admin_email: adminEmail,
                product_name: nombreProducto,
                current_stock: stockActual
            };

            if(EMAILJS_TEMPLATE_ADMIN !== "TU_TEMPLATE_ADMIN") {
                emailjs.send(EMAILJS_SERVICE_id, EMAILJS_TEMPLATE_ADMIN, adminParams)
                    .then(function(response) {
                        console.log(`Alerta enviada automáticamente al correo del administrador.`);
                    }, function(error) {
                        console.error('Error en EmailJS al enviar al Admin:', error);
                    });
            }
        }

        // Corrección aplicada: Se eliminó el bloque condicional de simulación
        function marcarComoEntregado(orderId) {
            const orden = ordenesCompra.find(o => o.id === orderId);
            if (orden) {
                orden.estado = "Entregado";
                saveData();
                renderAdminData(false); 

                const entregaParams = {
                    client_name: orden.cliente,
                    client_email: orden.email || "correo@desconocido.com",
                    product_name: orden.producto,
                    quantity: orden.cantidad,
                    total_price: orden.total,
                    delivery_address: orden.direccion
                };

                emailjs.send(EMAILJS_SERVICE_id, EMAILJS_TEMPLATE_ENTREGA, entregaParams)
                    .then(function(response) {
                        alert(`La orden de "${orden.cliente}" ha sido cambiada a Entregado.\n¡Correo de finalización enviado a ${entregaParams.client_email}!`);
                    }, function(error) {
                        alert(`La orden se cambió a Entregado, pero ocurrió un error con EmailJS al enviar la notificación a ${entregaParams.client_email}.`);
                        console.error("Detalles del error de entrega EmailJS:", error);
                    });
            }
        }

        function renderAdminData(triggerEmails = false) {
            const invBody = document.getElementById('inventory-table-body');
            if (invBody) invBody.innerHTML = '';
            
            const stockBody = document.getElementById('stock-table-body');
            if (stockBody) stockBody.innerHTML = '';

            const alertContainer = document.getElementById('stock-alerts-container');
            if (alertContainer) alertContainer.innerHTML = '';

            inventario.forEach(prod => {
                const precioPublico = prod.precioTienda * 1.20;
                
                if (invBody) {
                    const trInv = document.createElement('tr');
                    trInv.innerHTML = `
                        <td><span style="font-size:1.2rem;">${prod.icono}</span> <strong>${prod.nombre}</strong></td>
                        <td><input type="number" id="edit-price-${prod.id}" value="${prod.precioTienda}" style="width:70px;"></td>
                        <td>$${precioPublico.toFixed(2)}</td>
                        <td>${prod.stock} uds</td>
                        <td><input type="number" id="edit-stock-${prod.id}" value="${prod.stock}" style="width:60px;"></td>
                        <td><button class="btn btn-success" style="padding:5px 10px; font-size:0.8rem;" onclick="updateProductData(${prod.id})">Guardar</button></td>
                    `;
                    invBody.appendChild(trInv);
                }

                if (stockBody) {
                    const trStock = document.createElement('tr');
                    trStock.innerHTML = `<td>${prod.icono} ${prod.nombre}</td><td>${prod.stock} unidades</td>`;
                    stockBody.appendChild(trStock);
                }

                if(prod.stock < 10) {
                    if (alertContainer) {
                        const alertDiv = document.createElement('div');
                        alertDiv.className = 'stock-alert';
                        alertDiv.textContent = `⚠️ NOTIFICACIÓN: El producto "${prod.nombre}" está en stock mínimo (${prod.stock} unidades).`;
                        alertContainer.appendChild(alertDiv);
                    }
                    if (triggerEmails) {
                        enviarCorreoAdminStockBajo(prod.nombre, prod.stock);
                    }
                }
            });

            const ordersBody = document.getElementById('orders-table-body');
            if (ordersBody) {
                ordersBody.innerHTML = '';
                ordenesCompra.forEach(ord => {
                    const isEntregado = ord.estado === 'Entregado';
                    
                    const trOrd = document.createElement('tr');
                    trOrd.innerHTML = `
                        <td>${ord.cliente}</td>
                        <td>${ord.producto}</td>
                        <td>${ord.cantidad}</td>
                        <td>${ord.total}</td>
                        <td>${ord.direccion}</td>
                        <td style="color: ${isEntregado ? 'green' : 'orange'}; font-weight:bold;">${ord.estado}</td>
                        <td>
                            ${isEntregado 
                                ? '<span>✅ Finalizado</span>' 
                                : `<button class="btn btn-primary" style="padding:4px 8px; font-size:0.75rem;" onclick="marcarComoEntregado(${ord.id})">Entregado</button>`
                            }
                        </td>
                    `;
                    ordersBody.appendChild(trOrd);
                });
            }
        }
